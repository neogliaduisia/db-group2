# Модель прецедентів

## Загальна схема

<center style="
    border-radius: 4px;
    border: 1px solid #cfd7e6;
    padding: 16px;"
>

@startuml
  actor "Користувач" as User
  actor "Менеджер" as Manager
  actor "Адміністратор" as Admin
  
  Manager --u-|> User
  Admin -u-|> Manager
  
  usecase "<b>AccountSettings</b>\nУправління обліковим записом" as AccountSettings
  usecase "<b>NotificationSettings</b>\nУправління сповіщеннями" as NotificationSettings
  usecase "<b>TaskSettings</b>\nУправління задачами" as TaskSettings
  usecase "<b>WriteToSupport</b>\nНаписати в підтримку" as WriteToSupport
  usecase "<b>ProjectSettings</b>\nУправління проєктом" as ProjectSettings
  usecase "<b>TeamSettings</b>\nУправління командою" as TeamSettings
  usecase "<b>BoardManage</b>\nУправління дошкою" as BoardManage
  usecase "<b>ProjectTemplateSettings</b>\nУправління шаблоном проєкту" as ProjectTemplateSettings
  usecase "<b>ProjectWorkflowMonitor</b>\nМоніторинг робочого процесу проєкту" as ProjectWorkflowMonitor
  usecase "<b>DevToolsSettings</b>\nУправління інструментами розробника" as DevToolsSettings
  usecase "<b>PermissionsSettings</b>\nУправління дозволами" as PermissionsSettings
  usecase "<b>SestemConfigSettings</b>\nУправління налаштуваннями системи" as SestemConfigSettings
  
  User -u-> NotificationSettings
  User -u-> AccountSettings
  User -l-> WriteToSupport
  User -r-> TaskSettings
  
  Manager -l-> ProjectTemplateSettings
  Manager -r-> TeamSettings
  Manager -u-> ProjectWorkflowMonitor
  Manager -u-> DevToolsSettings
  TaskSettings -[hidden]-> DevToolsSettings
  WriteToSupport -[hidden]-> ProjectWorkflowMonitor
  
  Manager -d-> BoardManage
  Manager -d-> ProjectSettings
  ProjectSettings <-[hidden]- PermissionsSettings
  BoardManage -[hidden]-> SestemConfigSettings
  Admin -l-> PermissionsSettings
  Admin -r-> SestemConfigSettings
  
@enduml

</center>

@startuml
